$$ x_{1} = x_{0} + v_{0}\delta t + \frac{1}{2}a\delta t^{2}$$ $$ v_{1} = v_{0} + a\delta t$$

$$ x_{2} = x_{1} + v_{1}\delta t + \frac{1}{2}a\delta t^{2}$$ $$ v_{2} = v_{1} + a\delta t$$

$$x_{2} = (x_{0} + v_{0}\delta t + \frac{1}{2}a\delta t^{2}) + (v_{0} + a\delta t)\delta t + \frac{1}{2}a\delta t^{2}$$ $$ v_{2} = v_{0} + a\delta t + a\delta t$$

$$x_{2} = x_{0} + 2v_{0}\delta t + 2a\delta t^{2}$$ $$v_{2} = v_{0} + 2a\delta t$$

$$\delta t = \frac{t_{i}}{i}$$

$$x_{2} = x_{0} + 2v_{0}\frac{t_{2}}{2} + 2a\frac{t_{2}}{2}^{2}$$ $$v_{2} = v_{0} + 2a\frac{t_{2}}{2}$$

$$x_{2} = x_{0} + v_{0}t_{2} + \frac{1}{2}a{t_{2}}^{2}$$ $$v_{2} = v_{0} + at_{2}$$

$$x_{i+1} = x_{0} + v_{0}t_{i+1} + \frac{1}{2}a{t_{i+1}}^{2}$$ $$v_{i+1} = v_{0} + at_{i+1}$$

Question 2)

Program that models the numerical and analytic solution for a particle with acceleration = -K*x. The program numerically plots the particle using a simple integration scheme. The amplitude factor is found as the percent change in the amplitude between peaks, and was used in finding a dt value that kept it just below 1%. Afterwards, the energy error was (E(t) - E(0)) was plotted for 0 < t <= 4pi. The period was found by measuring the distance between zeros on the numerical function. The period was then found for velocities = .25, .5, 1, 2, 4, 8.

import sys, math
import matplotlib.pyplot as plt

# FUNCTION DEFINITIONS ==============================

# Interperpolation Function
def interp(x0, y0, x1, y1, x=None, y=None):
    if y == None:
        return y0 + (y1 - y0) * (x - x0) / (x1 - x0)
    elif x == None:
        return (y-y0) * (x1 - x0) / (y1 - y0) + x0

# Particle Acceleration
def acc(x, v):
    return -K * x 

# Potential - calculated from acceleration
def potential(x):
    return .5 * K * x**2

# Total Energy
def energy(x, v):
    Etot = potential(x) + 0.5*v*v
    return Etot

# Analytic Solution - from lecture
def analytic_soln(t):
    x = .5 * V0 * math.sin(2*t)
    v = V0 * math.cos(2*t)
    return x, v

# Main Function that increments the numerical position
def take_a_step(x, v, t, dt):
    
    # Set the acceleration.
    a = acc(x, v)

    # Take the time step.
    xp = x
    x += v*dt + 0.5*a*dt*dt
    v += a*dt
    t += dt
    
    # Future values to determine amplitude
    af = acc(x, v)
    xf = x + (v*dt + .5*af*dt*dt)
    
    return xp, x, xf, v, t

# Function that checks for and lists interpolated zeros
def check_zero(xp, x, t):
    if len(zeros_list) != 3: #Checks for not three because the zeros list only needs three elements.
        if (xp < 0) and (x >= 0):
            zeros_list.append(interp(t-dt, xp, t, x, y=0))
        if (xp > 0) and (x <= 0):
            zeros_list.append(interp(t-dt, xp, t, x, y=0))

# Function that checks for and lists the peaks
def check_amplitude(xp, x, xf, t):
    if (xp < x) and (xf <= x):
        amplitude_list.append(x)
        amplitude_time_list.append(t)

# Function that initializes the lists used in every answer, made to make the code less repetitive
def initialize_list():
    time_list = []
    numerical_list = []
    analytic_list = []
    energy_list = []
    zeros_list = []
    amplitude_list = []
    amplitude_time_list = []
    return [time_list, numerical_list, analytic_list, energy_list, 
           zeros_list, amplitude_list, amplitude_time_list]

# Function that initializes/resets the constants used in every answer, makes the code shorter
def initialize_system(X0, V0, DT):
    t    = 0
    x    = X0
    xp   = x
    xf   = 0
    v    = V0
    dt   = DT
    tmax = 4*math.pi
    return t, x, xp, xf, v, dt, tmax


# MAIN PROGRAM ===================================== 
# Define constants and initial conditions

X0 = 0
V0_list = [1]
K  = 4
DT = .05


for V0 in V0_list:    
    # Initialize the system.
    t, x, xp, xf, v, dt, tmax = initialize_system(X0, V0, DT)

    # Initialize Plotting Lists
    time_list,numerical_list,analytic_list,energy_list,zeros_list,amplitude_list,amplitude_time_list = initialize_list()

    # initial energy
    E0 = energy(x, v)
    dEmax = 0
    while t <= tmax:
        
        # Appends the time, numerical, and analytic values
        time_list.append(t)
        numerical_list.append(x)
        analytic_list.append(analytic_soln(t)[0])
        check_zero(xp, x, t)
        check_amplitude(xp, x, xf, t)
        (xp, x, xf, v, t) = take_a_step(x, v, t, dt)
        energy_list.append(energy(x, v) - E0)

    amplitude_factor = 0
    for i in range(len(amplitude_list)-1): # Amplitude factor is the greatest percent change
        temp_factor = (amplitude_list[i+1] - amplitude_list[i]) * 100 / (amplitude_list[i])
        amplitude_factor = temp_factor if (temp_factor > amplitude_factor) else amplitude_factor 
        # replaces amplitude factor is the new one is greater than the old one.
    
    plt.figure(1)
    plt.plot(time_list, numerical_list, 'r-', label = 'Numerical')
    plt.plot(time_list, analytic_list, 'b--', label = 'Analytic')
    plt.plot(amplitude_time_list, amplitude_list, 'go', label = 'Amplitude')
    plt.title('Numerical and Analytic Plots VS Time.')
    plt.xlabel('t')
    plt.ylabel('x')
    plt.legend()
    plt.legend(bbox_to_anchor=(1.3,1))
    
    print('Part A & B)')
    print('The amplitude factor for dt = {}: {:.2f}%'.format(dt, amplitude_factor))
