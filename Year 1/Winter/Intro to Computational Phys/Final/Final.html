def interp(x0, y0, x1, y1, x=None, y=None):
        if y == None:
            return y0 + (y1 - y0) * (x - x0) / (x1 - x0)
        elif x == None:
            return (y-y0) * (x1 - x0) / (y1 - y0) + x0

def acc(x, y, vx, vy, t):
    global alpha, beta, g, GM, thrust
    
    r_temp = sqrt(x**2 + y**2)
    v_abs  = sqrt(vx**2 + vy**2)
    
    ax_grav = -GM * x / r_temp**3
    ay_grav = -GM * y / r_temp**3
    
    ax_frict = -beta * v_abs * vx
    ay_frict = -beta * v_abs * vy
    
    ax_thrust = 10 * g * vx / v_abs
    ay_thrust = 10 * g * vy / v_abs
    
    if thrust:
        if t <= 75.5:
            return ax_grav + ax_thrust + ax_frict, ay_grav + ay_thrust + ay_frict
        else:
            return ax_grav + ax_frict, ay_grav + ay_frict
    else:
        return ax_grav + ax_frict, ay_grav + ay_frict
    
def take_a_step(x, y, vx, vy, t, dt):

        ax,ay = acc(x, y, vx, vy, t)

        # Predict: 
        x  += vx*dt + 0.5*ax*dt*dt
        y  += vy*dt + 0.5*ay*dt*dt
        vx += ax*dt
        vy += ay*dt

        # Correct: 
        ax1,ay1 = acc(x, y, vx, vy, t)

        vx += 0.5*(ax1-ax)*dt
        vy += 0.5*(ay1-ay)*dt

        t += dt

        return x,y,vx,vy,t

def acc(x, y):
    # inverse-square law
    r = sqrt(x*x+y*y)
    return -GM*x/pow(r, 3), -GM*y/pow(r, 3) 		


def take_a_step(x, y, vx, vy, t, dt):
    # Predictor:
    ax, ay = acc(x, y)

    x  += vx*dt + 0.5*ax*dt*dt
    y  += vy*dt + 0.5*ay*dt*dt
    vx += ax*dt
    vy += ay*dt
    t  += dt

    # Corrector:
    ax1, ay1 = acc(x, y)

    vx += 0.5*(ax1-ax)*dt
    vy += 0.5*(ay1-ay)*dt

    return x,y,vx,vy,t

def acc(x, y, vx, vy, t):
    
    
    x_earth = x - 0
    y_earth = y - 0
    
    x_moon = x - Moon_distance[0]
    y_moon = y - Moon_distance[1]
    
    r_earth = sqrt(x_earth**2 + y_earth**2)
    r_moon  = sqrt(x_moon**2 + y_moon**2)

    ax_earth = -GM * x_earth / r_earth**3
    ay_earth = -GM * y_earth / r_earth**3

    ax_moon = -G*Moon_M * x_moon / r_moon**3
    ay_moon = -G*Moon_M * y_moon / r_moon**3
    
    ax_sun = 0
    ay_sun = 2 * A * y 

    return ax_earth + ax_moon + ax_sun, ay_earth + ay_moon + ay_sun
